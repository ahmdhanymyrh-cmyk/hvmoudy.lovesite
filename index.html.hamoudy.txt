  <script>
    const correctPassword = 'Love';
    const overlay = document.getElementById('overlay');
    const unlockBtn = document.getElementById('unlockBtn');
    const passInput = document.getElementById('passwordInput');
    const errMsg = document.getElementById('errMsg');function unlock(){
  const v = passInput.value;
  if(v === correctPassword){
    overlay.style.display = 'none';
  } else {
    errMsg.style.display = 'block';
    passInput.classList.add('shake');
    setTimeout(()=>passInput.classList.remove('shake'),500);
  }
}

unlockBtn.addEventListener('click', unlock);
passInput.addEventListener('keydown', (e)=>{ if(e.key === 'Enter') unlock(); });

// --- floating hearts generator ---
const heartsContainer = document.querySelector('.hearts');
function makeHeart(){
  const i = document.createElement('i');
  const size = 12 + Math.random()*28;
  i.style.width = size + 'px';
  i.style.height = size + 'px';
  i.style.left = Math.random()*100 + '%';
  const delay = Math.random()*2;
  const dur = 6 + Math.random()*8;
  i.style.bottom = '-20px';
  i.style.opacity = 0.9;
  i.style.transform = 'rotate(45deg)';
  i.style.animation = `floatUp ${dur}s linear ${delay}s forwards`;
  heartsContainer.appendChild(i);
  setTimeout(()=>{ i.remove(); }, (delay+dur)*1000 + 500);
}
setInterval(makeHeart, 700);
